# Frontend Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the complete portfolio frontend as HTML/CSS/JS served by Flask, fetching all dynamic content from the existing API.

**Architecture:** `templates/index.html` holds all HTML structure; `static/css/style.css` holds all CSS (design system + per-section styles); `static/js/app.js` handles all API fetches and interactivity. Flask serves the page via a `GET /` route in `routes/frontend.py`. The design uses scroll-snap full-screen sections with a cobalt → deep violet gradient palette, light/dark theme toggle, and dusty rose for personality accents only.

**Tech Stack:** HTML5, CSS3 (custom properties, scroll-snap, gradient text, IntersectionObserver), vanilla JS (fetch API), Flask render_template

- [ ] **Step 6: Create `static/js/app.js`**

```javascript
/* ============================================================
   CONFIG
   ============================================================ */
const API_BASE = 'http://127.0.0.1:5000';

async function api(path, opts = {}) {
  const defaults = { headers: { 'Content-Type': 'application/json' } };
  const res = await fetch(API_BASE + path, { ...defaults, ...opts });
  if (!res.ok) throw new Error(`API ${res.status}: ${path}`);
  return res.json();
}

/* ============================================================
   THEME
   ============================================================ */
function initTheme() {
  const saved = localStorage.getItem('portfolio-theme') || 'dark';
  document.documentElement.setAttribute('data-theme', saved);
  updateThemeBtn(saved);
}

function toggleTheme() {
  const current = document.documentElement.getAttribute('data-theme');
  const next = current === 'dark' ? 'light' : 'dark';
  document.documentElement.setAttribute('data-theme', next);
  localStorage.setItem('portfolio-theme', next);
  updateThemeBtn(next);
}

function updateThemeBtn(theme) {
  const btn = document.getElementById('theme-btn');
  if (btn) btn.textContent = theme === 'dark' ? '\u2600 Light' : '\u25D1 Dark';
}

/* ============================================================
   TOAST
   ============================================================ */
function showToast(msg, ms = 2800) {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.classList.add('show');
  setTimeout(() => t.classList.remove('show'), ms);
}

/* ============================================================
   SCROLL REVEAL
   ============================================================ */
function initScrollReveal() {
  const els = document.querySelectorAll('.reveal, .reveal-l, .reveal-r');
  const obs = new IntersectionObserver(
    (entries) => entries.forEach(e => { if (e.isIntersecting) e.target.classList.add('visible'); }),
    { threshold: 0.12 }
  );
  els.forEach(el => obs.observe(el));
}

/* ============================================================
   MAIN
   ============================================================ */
async function init() {
  initTheme();
  initScrollReveal();

  try {
    const [about, facts, projects, books, songs] = await Promise.all([
      api('/about'),
      api('/fun-fact/all'),
      api('/projects'),
      api('/books'),
      api('/songs'),
    ]);

    initHero(about);
    initAbout(about);
    initFunFacts(facts);
    initProjects(projects);
    initBooks(books);
    initSongs(songs);
  } catch (e) {
    console.error('API fetch failed:', e);
  }

  initContact();
  initBooksForm();
  initSongsForm();
}

document.addEventListener('DOMContentLoaded', init);
```

/* ============================================================

const ALLOWED_EMOJIS = ['\uD83D\uDE02', '\uD83D\uDC80', '\uD83D\uDE2D', '\uD83E\uDEF6', '\uD83D\uDC40'];

function buildReactions(fact) {

  div.className = 'reactions';
  ALLOWED_EMOJIS.forEach(emoji => {
    const btn = document.createElement('div');
    btn.className = 'react';
    const count = (fact.reactions && fact.reactions[emoji]) || 0;
    btn.innerHTML = `${emoji} <span class="react-ct">${count}</span>`;
    btn.addEventListener('click', async () => {
      try {
        const data = await api(`/fun-fact/${fact.id}/react`, {
          method: 'POST',
          body: JSON.stringify({ emoji }),
        });
        btn.querySelector('.react-ct').textContent =
          (data.reactions && data.reactions[emoji]) || 0;
      } catch (e) {
        console.error('reaction failed:', e);
      }
    });
    div.appendChild(btn);
  });
  return div;
}

function initFunFacts(facts) {

    const card = document.createElement('div');
    card.className = 'fact-card reveal';
    const p = document.createElement('p');
    p.className = 'fact-text';
    p.textContent = `"${fact.fact}"`;
    card.appendChild(p);
    card.appendChild(buildReactions(fact));
    grid.appendChild(card);
  });
